18      /**
19       * Main application class.
20       * @param args arguments.
21       */
22      public static void main(String[] args) {
23          GameBoard view = new GameBoard();
24          Game g = new Game(
25                  view,
26                  new TetrominoRandomizer(),
27                  new Board(new boolean[20][10])
28          );
29          g.setMovement(new Movement(g));
30          g.nextTetromino();
31  
32          Window w = new Window(view);
33          SwingUtilities.invokeLater(w);
34  
35          Timer t = new Timer();
36  
37          t.scheduleAtFixedRate(new GameTickTask(g), 600, 600);
38      }
